## FRAMER MOTION — ANIMATION RULES

Create `components/RevealWrapper.tsx`:
```typescript
'use client'
import { motion, useInView } from 'framer-motion'
import { useRef } from 'react'

interface RevealWrapperProps {
  children: React.ReactNode
  delay?: number
  direction?: 'up' | 'left' | 'right'
  className?: string
}

export function RevealWrapper({ children, delay = 0, direction = 'up', className }: RevealWrapperProps) {
  const ref = useRef(null)
  const isInView = useInView(ref, { once: true, margin: '-80px' })
  
  const variants = {
    hidden: {
      opacity: 0,
      y: direction === 'up' ? 28 : 0,
      x: direction === 'left' ? -36 : direction === 'right' ? 36 : 0,
    },
    visible: { opacity: 1, y: 0, x: 0 }
  }

  return (
    <motion.div
      ref={ref}
      variants={variants}
      initial="hidden"
      animate={isInView ? 'visible' : 'hidden'}
      transition={{ duration: 0.7, ease: [0.25, 0.1, 0.25, 1], delay }}
      className={className}
    >
      {children}
    </motion.div>
  )
}
```

Use stagger on card grids:
```typescript
const containerVariants = {
  hidden: {},
  visible: { transition: { staggerChildren: 0.1 } }
}
const cardVariants = {
  hidden: { opacity: 0, y: 24 },
  visible: { opacity: 1, y: 0, transition: { duration: 0.6, ease: [0.25, 0.1, 0.25, 1] } }
}
```

Page-level entry animations for hero content (badge → h1 → subtitle → buttons, each staggered 0.2s).
